$('input[type="file"].thumb-available').each(function(index, node) { const $node = $(node); $node.on('change', function () { const thumbSelector = $node.attr('data-thumb-selector'); if (typeof (FileReader) != "undefined") { const evalStr = '$node.' + thumbSelector; var $imageHolder = eval(evalStr); $imageHolder.empty(); var reader = new FileReader(); reader.onload = function (e) { $("", { "src": e.target.result, "class": "max-w-sm block rounded border-2 border-gray-500" }).appendTo($imageHolder); } reader.readAsDataURL($(this)[0].files[0]); } else { alert("This browser does not support FileReader."); } }); });
